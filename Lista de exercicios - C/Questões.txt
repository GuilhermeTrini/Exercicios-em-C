int main()
{
    int tempo, horas, minutos, segundos;
    scanf("%d", &tempo);
    horas = tempo / 3600;
    minutos = (tempo - (horas * 3600)) / 60;
    segundos = tempo - (minutos * 60) - (horas * 3600);
    printf("%02d:%02d:%02d", horas, minutos, segundos);
    return 0;
}
